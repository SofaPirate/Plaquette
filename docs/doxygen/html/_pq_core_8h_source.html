599void Engine::preStep() {
600 // Update every component.
601 for (size_t i=_unitsBeginIndex; i != _unitsEndIndex; i++) {
602 units()[i]->step();
603 }
604
605 // Look for events.
606 _eventManager.step();
607}
608
609void Engine::postStep() {
610 // Increment step.
611 _nSteps++;
612
613 // Calculate true sample rate.
614 _updateGlobalMicroSeconds();
615 uint32_t diffTime = _totalGlobalMicroSeconds.micros32.base - _microSeconds.micros32.base;\
616 float trueSampleRate = (diffTime ? MICROS_PER_SECOND / diffTime : PLAQUETTE_MAX_SAMPLE_RATE);
617
618 // If we are in auto sample mode OR if the target sample rate is too fast for the "true" sample rate
619 // then we should just assign the true sample rate.
620 if (autoSampleRate() || trueSampleRate < _targetSampleRate) {
621 _setSampleRate(trueSampleRate);
622 _microSeconds = _totalGlobalMicroSeconds;
623 }
624
625 // Otherwise: Wait in order to synchronize seconds with real time.
626 else {
627 uint32_t startTime = _microSeconds.micros32.base;
628 micro_seconds_t targetTime = _microSeconds;
629 targetTime.micros32.base += (uint32_t)(MICROS_PER_SECOND/_targetSampleRate + 0.5f); // rounded
630
631 if (targetTime.micros32.base < startTime) { // overflow
632 targetTime.micros32.overflows++;
633 while (_updateGlobalMicroSeconds().micros64 < targetTime.micros64); // wait
634 _microSeconds = targetTime;
635 }
636 else {
637 while (_updateGlobalMicroSeconds().micros32.base < targetTime.micros32.base); // wait
638 _microSeconds.micros32.base = targetTime.micros32.base; // not the exact "true" time but more accurate for computations
639 }
640 _setSampleRate(_targetSampleRate);
641 }
642}

648void Engine::step() {
649 if (_firstRun) {
650 postBegin();
651 _firstRun = false;
652 }
653 else
654 postStep();
655
656 // Do the pre-step.
657 preStep();
658}
