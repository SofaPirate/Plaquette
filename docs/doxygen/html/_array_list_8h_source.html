653 void bubbleSort(bool (*comparator)(T, T)) {
654 for (size_t i = 0; i < count - 1; ++i) {
655 for (size_t j = 0; j < count - i - 1; ++j) {
656 if (comparator(array[j], array[j + 1])) {
657 T temp = array[j];
658 array[j] = array[j + 1];
659 array[j + 1] = temp;
660 }
661 }

720 delete[] array;

741 int partition(bool (*comparator)(T, T), int low, int high) {
742 T pivot = array[high];
743 int i = (low - 1);
744 for (int j = low; j <= high - 1; j++) {
745 if (comparator(array[j], pivot)) {
746 i++;
747 swap(&array[i], &array[j]);
748 }
749 }
750 swap(&array[i + 1], &array[high]);
751 return (i + 1);
752 }

765 void quickSortHelper(bool (*comparator)(T, T), int low, int high) {
766 if (low < high) {
767 int pi = partition(comparator, low, high);
768 quickSortHelper(comparator, low, pi - 1);
769 quickSortHelper(comparator, pi + 1, high);
770 }
771 }
